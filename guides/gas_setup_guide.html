1Google スプレッドシートを作成する

sheets.new を開いて新しいスプレッドシートを作成してください。

シート名は何でもOKです（例：「GC採用管理DB」）。

💡 シートのタブ名（下部）は変えなくてOKです。スクリプトが自動で Candidates・Jobs・TeamMembers シートを作成します。

作成したら、スプレッドシートを 社内のHRメンバーと共有（閲覧権限）しておくと、データをスプレッドシートで直接確認できます。

2Apps Script エディタを開く

スプレッドシートのメニューから：

「拡張機能」→「Apps Script」 をクリック

💡 Apps Script のエディタが別タブで開きます。左側に コード.gs というファイルがあります。

既存の内容（function myFunction() {} など）はすべて削除して、次のステップのコードを貼り付けます。

3スクリプトをコピー&ペーストする

以下のコードをまるごとコピーして、Apps Script エディタに貼り付けてください。

⚠️ 既存のコードはすべて削除してから貼り付けてください。
// ================================================================
// Green Carbon HR Tools - GAS バックエンド (完全版)
// ================================================================
// 対応API:
//   GET  ?action=getCandidates    → 候補者一覧取得
//   GET  ?action=getJobs          → 求人一覧取得
//   GET  ?action=getTeamMembers   → チームメンバー取得
//   POST action=saveCandidates    → 候補者データ保存
//   POST action=saveJob           → 求人追加
//   POST action=saveTeamMembers   → チームメンバー保存
// ================================================================

function doGet(e) {
  try {
    const action = e.parameter.action;
    const ss = SpreadsheetApp.getActiveSpreadsheet();
    let result;

    if (action === 'getCandidates') {
      result = readSheet(ss, 'Candidates',
        ['evaluations', 'interviews', 'memos', 'reviewers']);

    } else if (action === 'getJobs') {
      result = readSheet(ss, 'Jobs', []);

    } else if (action === 'getTeamMembers') {
      result = readSheet(ss, 'TeamMembers', []);

    } else {
      result = { error: 'Unknown action: ' + action };
    }

    return buildResponse(result);

  } catch (err) {
    return buildResponse({ error: err.message });
  }
}

function doPost(e) {
  try {
    const body = JSON.parse(e.postData.contents);
    const ss = SpreadsheetApp.getActiveSpreadsheet();

    if (body.action === 'saveCandidates') {
      const keys = [
        'id', 'name', 'nationality', 'position', 'stage',
        'priority', 'assignee', 'channel', 'email', 'appliedDate',
        'dept', 'evaluations', 'interviews', 'memos', 'reviewers',
        'activityDate', 'evaluator', 'evalDate'
      ];
      writeSheet(ss, 'Candidates', keys, body.data,
        ['evaluations', 'interviews', 'memos', 'reviewers']);
      return buildResponse({ ok: true, saved: body.data.length });

    } else if (body.action === 'saveJob') {
      const sheet = getOrCreateSheet(ss, 'Jobs');
      const keys = ['id', 'title', 'dept', 'target', 'status', 'desc'];
      if (sheet.getLastRow() === 0) sheet.appendRow(keys);
      sheet.appendRow(keys.map(k => body.data[k] || ''));
      return buildResponse({ ok: true });

    } else if (body.action === 'saveTeamMembers') {
      const keys = ['id', 'name', 'role', 'initial', 'color'];
      writeSheet(ss, 'TeamMembers', keys, body.data, []);
      return buildResponse({ ok: true, saved: body.data.length });

    } else {
      return buildResponse({ error: 'Unknown action' });
    }

  } catch (err) {
    return buildResponse({ error: err.message });
  }
}

// CORS プリフライト対応
function doOptions(e) {
  return ContentService
    .createTextOutput('')
    .setMimeType(ContentService.MimeType.TEXT);
}

// ── ヘルパー関数 ──────────────────────────────────────────────────

function buildResponse(data) {
  return ContentService
    .createTextOutput(JSON.stringify(data))
    .setMimeType(ContentService.MimeType.JSON);
}

function getOrCreateSheet(ss, name) {
  return ss.getSheetByName(name) || ss.insertSheet(name);
}

function readSheet(ss, sheetName, jsonFields) {
  const sheet = getOrCreateSheet(ss, sheetName);
  const data = sheet.getDataRange().getValues();
  if (data.length < 2) return [];
  const headers = data[0];
  return data.slice(1).map(row => {
    const obj = {};
    headers.forEach((h, i) => {
      if (jsonFields.includes(h)) {
        try { obj[h] = row[i] ? JSON.parse(row[i]) : []; }
        catch(e) { obj[h] = []; }
      } else {
        obj[h] = row[i];
      }
    });
    return obj;
  }).filter(row => row.id); // 空行除去
}

function writeSheet(ss, sheetName, keys, dataArr, jsonFields) {
  if (!dataArr || !dataArr.length) return;
  const sheet = getOrCreateSheet(ss, sheetName);
  sheet.clearContents();
  sheet.appendRow(keys);
  const rows = dataArr.map(item =>
    keys.map(k => {
      const v = item[k];
      if (jsonFields.includes(k)) return JSON.stringify(v || []);
      return (v === undefined || v === null) ? '' : v;
    })
  );
  if (rows.length) {
    sheet.getRange(2, 1, rows.length, keys.length).setValues(rows);
  }
}
✅ reviewers フィールド（チームチェック）も含む完全版です。コピーしたらそのまま使えます。

貼り付けたら Ctrl+S（⌘+S）で保存してください。

4ウェブアプリとしてデプロイする

Apps Script エディタの右上：

「デプロイ」→「新しいデプロイ」 をクリック

⚙️ 設定画面が開きます。以下の通りに設定してください：
設定項目設定値
種類の選択ウェブアプリ
次のユーザーとして実行自分（your-email@...）
アクセスできるユーザー全員（Anonymous を含む）
⚠️「アクセスできるユーザー：全員」にしないと、ツールからデータを取得できません。スプレッドシート自体は共有設定を変えなくてOKです。

「デプロイ」をクリック → 権限の確認画面が出たら「アクセスを許可」→ ウェブアプリのURLが表示されます。このURLをコピーしてください。

5URLをツールに設定する

gc_recruiting.html をテキストエディタで開き、先頭付近の GAS_URL を書き換えます：

// ファイル先頭付近（約10行目）
const GAS_URL = 'https://script.google.com/macros/s/★ここに新しいURLを貼る★/exec';
